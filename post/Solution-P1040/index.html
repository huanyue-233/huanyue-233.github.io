    显然,中序遍历是 左 根 右 的顺序

    因此,如果第iii号元素是[l,r][l,r][l,r]的根,那么iii左边的数构成了其左子树,右边的数构成了其右子树

    具体的,设f[l][r]f[l][r]f[l][r]表示[l,r][l,r][l,r]构成子树的最大加分,设rootrootroot表示示最大加分时[l,r][l,r][l,r]的根节点.

    那么显然有f[l][r]f[l][r]f[l][r]=f[l][root−1]∗f[root+1][r]+v[root]f[l][root-1]*f[root+1][r]+v[root]f[l][root−1]∗f[root+1][r]+v[root]

    这就像区间dp,满足一个大区间由小区间转移过来.

    首先,初始状态f[i][i]=v[i]f[i][i]=v[i]f[i][i]=v[i]

    1.枚举区间长度(从小到大)

    2.枚举区间左节点

    3.枚举断点

    4.取max,记录root

#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#include <queue>
#include <utility>
#include <vector>
#define rgi register
typedef long long ll;
typedef unsigned long long ull;
using namespace std;

const int N = 50;
const int inf = 0x3f3f3f3f;

inline void write(register int x)
{
    if (x < 0)
    {
        putchar('-'), x = -x;
    }
    if (x >= 10)
    {
        write(x / 10);
    }
    putchar('0' + x % 10);
}

inline int read()
{
    int x = 0, f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9')
    {
        if (ch == '-')
            f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9')
    {
        x = ((x << 3) + (x << 1) + (ch ^ 48));
        ch = getchar();
    }
    return x * f;
}

int n, v[39], f[47][47], i, j, k, root[49][49];
void print(int l, int r)
{
    if (l > r)
        return;
    if (l == r)
    {
        write(l);
        putchar(' ');
        return;
    }
    write(root[l][r]);
    putchar(' ');
    print(l, root[l][r] - 1);
    print(root[l][r] + 1, r);
}
int main()
{
    n = read();
    for (i = 1; i <= n; i++)
        v[i] = read();
    for (i = 1; i <= n; i++)
    {
        f[i][i] = v[i];
        f[i][i - 1] = 1;
    }
    for (i = n; i >= 1; i--)
        for (j = i + 1; j <= n; j++)
            for (k = i; k <= j; k++)
            {
                if (f[i][j] < (f[i][k - 1] * f[k + 1][j] + f[k][k]))
                {
                    f[i][j] = f[i][k - 1] * f[k + 1][j] + f[k][k];
                    root[i][j] = k;
                }
            }
    write(f[1][n]);
    putchar('\n');
    print(1, n);
    return 0;
}
