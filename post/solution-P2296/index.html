#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#include <queue>
#include <utility>
#include <vector>
#define rgi register
typedef long long ll;
typedef unsigned long long ull;
using namespace std;

inline void write(register int x)
{
    if (x < 0)
    {
        putchar('-'), x = -x;
    }
    if (x >= 10)
    {
        write(x / 10);
    }
    putchar('0' + x % 10);
}

inline int read()
{
    int x = 0, f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9')
    {
        if (ch == '-')
            f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9')
    {
        x = ((x << 3) + (x << 1) + (ch ^ 48));
        ch = getchar();
    }
    return x * f;
}
const int N = 10010, M = 200010;

vector<int> g[N], _g[N];
bool can[N], Can[N];
int dis[N];

const int QueueLen = 1e5;
const int Error_Number = -1; //队列长度
template <class T>
class Queue //最普通的队列
{
public:
    T Q[QueueLen];
    int head, tail;
    Queue()
    {
        head = tail = 0;
    }
    bool full();
    bool empty();
    void push(T x);
    void pop();
    T front();
};

template <class T>
bool Queue<T>::full()
{
    if (tail == QueueLen)
        return 1;
    return 0;
}

template <class T>
bool Queue<T>::empty()
{
    if (head == tail)
        return true;
    return false;
}

template <class T> //tail空
void Queue<T>::push(T x)
{
    if (full())
        return;
    Q[tail++] = x;
}

template <class T>
void Queue<T>::pop()
{
    if (empty())
        return;
    head++;
}

template <class T>
T Queue<T>::front() //队首
{
    if (empty())
        return Error_Number;
    return Q[head];
}
int main()
{
    int n = read(), m = read();
    for (rgi int i = 1; i <= m; i++)
    {
        int u = read(), v = read();
        g[u].push_back(v);
        _g[v].push_back(u);
    }
    int s = read(), t = read();
    can[t] = 1;
    Queue<int> q;
    q.push(t);
    while (!q.empty())
    {
        int now = q.front();
        q.pop();
        for (vector<int>::iterator iter = _g[now].begin(); iter != _g[now].end(); iter++)
        {
            if (!can[*iter])
            {
                q.push(*iter);
                can[*iter] = 1;
            }
        }
    }
    if (!can[s])
    {
        puts("-1");
        return 0;
    }
    // cout << "!!!";
    for (rgi int i = 1; i <= n; i++)
    {
        if (can[i])
        {
            Can[i] = 1;
            for (vector<int>::iterator iter = g[i].begin(); iter != g[i].end(); iter++)
            {
                if (!can[*iter])
                {
                    Can[i] = 0;
                    break;
                }
            }
        }
    }
    if (!Can[s])
    {
        puts("-1");
        return 0;
    }
    // cout << "!!!";
    dis[s] = 1;
    q.push(s);
    while (!q.empty())
    {
        int now = q.front();
        q.pop();
        if (now == t)
        {
            cout << dis[t] - 1;
            return 0;
        }
        for (vector<int>::iterator iter = g[now].begin(); iter != g[now].end(); iter++)
        {
            if (Can[*iter] && !dis[*iter])
            {
                dis[*iter] = dis[now] + 1;
                q.push(*iter);
            }
        }
    }
    cout << "-1";
    return 0;
}
